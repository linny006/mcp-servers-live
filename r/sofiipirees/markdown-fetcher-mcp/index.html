Install & configure

Claude Desktop config snippet

Add to your claude_desktop_config.json (~/Library/Application Support/Claude/claude_desktop_config.json on macOS, %APPDATA%\Claude\claude_desktop_config.json on Windows):

{
  "mcpServers": {
    "html-to-markdown-pro": {
      "command": "node",
      "args": [
        "dist/server.js",
        "--port",
        "3100",
        "--max-retries",
        "3",
        "--timeout",
        "10000"
      ],
      "env": {
        "CONVERSION_LIMIT": "500KB",
        "STRIP_SCRIPTS": "true",
        "PRESERVE_LINKS": "true",
        "RESPONSIVE_OUTPUT": "true",
        "OPENAI_KEY": "sk-your-key-here",
        "CLAUDE_KEY": "sk-ant-your-key-here"
      }
    }
  }
}
