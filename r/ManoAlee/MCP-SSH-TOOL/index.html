Install & configure

Claude Desktop config snippet

Add to your claude_desktop_config.json (~/Library/Application Support/Claude/claude_desktop_config.json on macOS, %APPDATA%\Claude\claude_desktop_config.json on Windows):

{
  "mcpServers": {
    "ssh-powershell-mcp": {
      "type": "stdio",
      "command": "C:\\ssh-mcp\\server\\.venv\\Scripts\\python.exe",
      "args": [
        "C:\\ssh-mcp\\server\\src\\ssh_connect\\server.py"
      ],
      "cwd": "C:\\ssh-mcp\\server",
      "env": {
        "SSH_HOST": "localhost",
        "CONNECTION_MODE": "powershell",
        "SSH_PORT": "22",
        "SSH_USERNAME": "",
        "SSH_PASSWORD": "",
        "PYTHONUNBUFFERED": "1"
      }
    }
  }
}
